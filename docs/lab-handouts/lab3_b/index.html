Part-A: Design a 3-to-8 binary decoder by instantiating only 2-to-1 multiplexers.#

Use below mentioned description of 2x1 Mux to describe Decoder#

2 x 1 MUX
library ieee;use ieee.std_logic_1164.all ;
entity mux_2_1 is port ( inp1, inp0 : in std_logic ; sel : in std_logic ;       outp  : out std_logic ) ;end entity ;
architecture rch_1_mux_2_1 of mux_2_1 is begin  outp <= inp1 when sel='1' else inp0 ;end rch_1_mux_2_1 ;
